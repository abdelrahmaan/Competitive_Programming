#include <bits/stdc++.h>
using namespace std;
const int MAX = 2e5 + 1e3 , OO = 1000000009;
int n , k  , i ;

    scanf("%d%d" , &n  , &k);
     for(int i = 0; i < n; ++i){
         for(int j = 0; j < n; ++j){
