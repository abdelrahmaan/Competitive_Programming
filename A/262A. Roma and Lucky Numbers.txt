#include <bits/stdc++.h>
using namespace std;
const int MAX = 2e5 + 1e3 , OO = 1000000009;
int n , k , lucky , len , ans;
string s;
int main(){
    scanf("%d%d" , &n , &k);
     for(int i = 0; i < n; ++i){
     cin >> s;
     len = s.size();
        for(int i = 0; i < len; ++i)
            if(s[i] == '4' || s[i] == '7'  ) lucky++;
        if(lucky <= k) ans++;
        lucky = 0;
        }
    printf("%d\n" , ans);
    return 0;
}
