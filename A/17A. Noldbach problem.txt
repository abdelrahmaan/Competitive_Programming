const int N = 1e6+6 , MAX = 2e5+5  , OO = 1000000009;
int sum[1007] , n  , k , ans , x , m;
    vector<bool> primes(1007 , 1);
void seive(){
    primes[0] = primes[1] = 0;
    for(int i = 2; i * i<= 1007 ; ++i)
        if(primes[i])
            for(int j = 2 * i; j <= 1007; j +=i)
                primes[j] = 0;
        }
int main(){
    scanf("%d%d" , &n , &k);
    seive();
     for(int i = 2; i <= n ; ++i)
        if(primes[i]){
            sum[m] = i;
            ++m;
     }
    for(int i = 1; i <= n ; ++i){
         x = sum[i] + sum[i-1] + 1;
         if(x <= n && primes[x]) ++ans;
    }
    puts((ans >= k ? "YES" : "NO"));
        return 0 ;
}
