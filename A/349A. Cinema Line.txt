int n   , A[N] , sum , n25 , n50;
int main() {
    scanf("%d" , &n );
    for(int i = 0 ; i < n ; ++i)    scanf("%d" , &A[i]);
    if(A[0] > 25)   return puts("NO") , 0;
    for(int i = 0 ; i < n ; ++i){
        if(A[i] == 25)              ++n25;
        else if (A[i] == 50)        ++n50 , --n25;
        else if (A[i] == 100){
            bool f = 0;
            if(n25 && n50)      --n25 , --n50 , f = 1;
            else if (n25 >= 3)  n25 -= 3 , f = 1;
            if (!f) return puts("NO") , 0;
        }
    }
    if(n25 < 0 || n50 < 0)           puts("NO");
    else                             puts("YES");
    return 0;
}
