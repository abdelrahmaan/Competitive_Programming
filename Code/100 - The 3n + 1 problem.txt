#include <bits/stdc++.h>
using namespace std;
int f3n_1(int n){
    int num =1;
    while(n != 1){
    if(n % 2 == 0)
          n /= 2;
        else
             n = 3 * n + 1;
            num++;
    }
            return num;
}
int main(){
int x = 0, y = 0 , xa, ya;
    while(cin>>x>>y){
        xa = x, ya = y;
        int maxi =0;
        if(x > y )
            swap(x , y);
        for(int i = x; i <= y; ++i)
             maxi = max(maxi , f3n_1(i));
        printf("%d %d %d\n" ,xa , ya , maxi);
    }
return 0;
}
